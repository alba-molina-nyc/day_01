bill = float(input('what was the total bill: ')) tip = int(input('what is the percentage you`d like to leave on this bill: ')) people = int(input('how many people will you be splitting this bill with: ')) tip_as_percent = tip /100 total_tip_amount = bill * tip_as_percent total_bill = bill + total_tip_amount bill_per_person = total_bill / people final_amount = round(bill_per_person) print(f'each person should pay {final_amount}')
